let table = document.getElementById('table');
const n = parseInt(prompt('Enter a number.', '10'));

const getSquare = n => n * n;

const getCube = n => n * n * n;

if (typeof n == "number" && n <= 10)

    for (let i = 0; i <= n; i++) {
        table.innerHTML += `
        <tr>
            <td>${i}</td>
            <td>${getSquare(i)}</td>
            <td>${getCube(i)}</td>
        </tr>
    `
    }

else alert('You need to input value between 0 & 10!');
