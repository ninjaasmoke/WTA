const elem = document.getElementById('text-anim');
let fs = 5;
let interval2;

const increase = () => {
    elem.innerText = "TEXT GROWING";
    elem.style.color = "#ff4454";
    elem.style.fontSize = `${fs}px`;
    fs++;
    if (fs > 50) {
        clearInterval(interval1);
        interval2 = setInterval(decrease, 100);
    }
}

let interval1 = setInterval(increase, 100);

const decrease = () => {
    elem.innerText = "TEXT SHRINKING";
    elem.style.color = "#5444ff";
    elem.style.fontSize = `${fs}px`;
    fs--;
    if (fs < 5) {
        clearInterval(interval2);
        interval1 = setInterval(increase, 100);
    }
}
